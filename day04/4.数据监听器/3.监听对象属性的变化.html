3。监听对象属性的变化 数据监听器支持监听对象中单个或多个属性的变化，示例语法如下: Component({ data: { rgba: { r: 0, g: 0, b: 0 }, color: '0,0,0' }, observers: { "rgba.r,rgba.g,rgba,b': function(r,g,b){ this.setdata({ color: `${r},${g},${b}` }) } }} Component({ observers: { "rgb.**': function(obj){ this.setdata({ color: `${obj.r},${obj.g},${obj.b}` }) } }}
